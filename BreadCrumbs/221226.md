# TypeScript

## getAncestorList

```TSX
// before
  const handleMenuItemClick = (menuItemId: number) => {
    setCurrentMenuItemIdList([
      ...getAncestorList(menuItemId).map((childrenItem) => childrenItem.id),
      menuItemId,
    ]);
  };

// after
  const handleMenuItemClick = (menuItemId: number) => {
    setCurrentMenuItemIdList(
      getAncestorList(menuItemId)
        .map((childrenItem) => childrenItem.id)
        .concat([menuItemId]),
    );
  };
```
